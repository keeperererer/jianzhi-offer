### 前序遍历
递归
```
function preOrderRecur(head){
	if(head == null){
		return;
	}
	console.log(head.value);
	preOrderRecur(head.left);
	preOrderRecur(head.right);
}
```
非递归
```
function preOrderRecur(head){
	if(head !== null){
		let stack = [];
		stack.push(head);
		while(stack.length !== null){
			head = stack.pop();
			console.log(head.value);
			if(head.right !== null){
				stack.push(head.right);
			}
			if(head.left !== null){
				stack.push(head.left);
			}
		}
	}
}
```

### 中序遍历
递归
```
function inOrderRecur(head){
	if(head == null){
		return ;
	}
	console.log(head.value);
	preOrderRecur(head.left);
	preOrderRecur(head.right);
}
```
非递归
```
function inOrderRecur(head){
	if(head !== null){
		let stack = [];
		while(stack.length !== null || head !== null){
			if(head !== null){
				stack.push(head);
				head = head.left;
			}else{
				head = stack.pop();
				console.log(head.value);
				head = head.right;
			}
		}	
	}
}
```
